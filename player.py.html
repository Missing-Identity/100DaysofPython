from turtle import Turtle

STARTING_POSITION = (0, -280)
MOVE_DISTANCE = 10
FINISH_LINE_Y = 280


class Player(Turtle):
    def __init__(self):
        super().__init__()
        pass

    def create_player(self):
        self.color("black")
        self.shape("turtle")
        self.penup()
        self.setposition(STARTING_POSITION)
        self.setheading(90)

    def move_player(self):
        self.forward(MOVE_DISTANCE)

    def player_reset(self):
        self.clear()
        self.reset()
