from turtle import Turtle

FONT = ("Courier", 24, "normal")


class Scoreboard():
    def __init__(self):
        self.score_text = Turtle()
        self.score_number = Turtle()

    def initial_text(self):
        self.score_text.penup()
        self.score_text.ht()
        self.score_text.setposition(-240, 240)
        self.score_text.write("Score: ", move=True, font=FONT)

    def score(self, level):
        self.score_number.clear()
        self.score_number.ht()
        self.score_number.penup()
        self.score_number.setposition(-150, 240)
        self.score_number.write(f"{(level-1)}", move=True, font=FONT)
