from turtle import Turtle
import random

COLORS = ["red", "orange", "yellow", "green", "blue", "purple"]


class CarManager:
    def __init__(self):
        self.number_of_cars = random.randint(5, 15)
        self.cars = []
        self.starting_move_distance = 5
        self.move_increment = 10
        self.level_tracker = 1
        pass

    def car_creator(self):
        car = Turtle("square")
        car.penup()
        car.color(random.choice(COLORS))
        initial_car_position_left = (-280, random.randint(-260, 260))
        initial_car_position_right = (280, random.randint(-260, 260))
        car.resizemode("user")
        car.shapesize(stretch_len=random.randint(1, 2), stretch_wid=1)
        side_picker = random.randint(0, 1)
        if side_picker == 0:
            car.setposition(initial_car_position_left)
            car.setheading(0)
        else:
            car.setposition(initial_car_position_right)
            car.setheading(180)
        self.cars.append(car)

    def car_move(self, level):
        if self.level_tracker == level:
            self.car_move_loop()
        else:
            self.starting_move_distance = self.starting_move_distance + self.move_increment
            self.level_tracker = level
            self.car_move_loop()

    def continuous_car_spawn(self):
        self.car_creator()

    def car_move_loop(self):
        for car in self.cars:
            car.forward(self.starting_move_distance)

    def car_reset(self):
        for car in self.cars:
            car.ht()
        self.cars.clear()
